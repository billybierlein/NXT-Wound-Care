Paste this and adjust imports to your paths. It includes:

the provider Select with undefined placeholder,

reset mapping for edit/new,

submit via PATCH,

invoiceStatus watcher to prompt for payment date.

import { useEffect, useMemo, useRef, useState } from "react";
import { useForm, Controller } from "react-hook-form";
import { z } from "zod";
import { zodResolver } from "@hookform/resolvers/zod";
import { Dialog, DialogContent, DialogHeader, DialogTitle } from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Form, FormField, FormItem, FormLabel, FormControl, FormMessage } from "@/components/ui/form";
import { Select, SelectTrigger, SelectValue, SelectContent, SelectItem } from "@/components/ui/select";
import { Input } from "@/components/ui/input";
// import your data hooks
// import { useProviders } from "@/hooks/useProviders";
// import { useTreatment } from "@/hooks/useTreatment";

type Provider = { id: number; name: string; active?: boolean };
type Treatment = {
  id?: number;
  providerId?: number | null;
  actingProvider?: string | null; // legacy
  invoiceStatus?: "open" | "paid" | "closed";
  paymentDate?: string | null; // ISO yyyy-mm-dd or null
  // ... add other fields you edit here
};

const schema = z.object({
  providerId: z.number().optional(),          // undefined when not selected
  invoiceStatus: z.enum(["open", "paid", "closed"]),
  paymentDate: z.string().optional().nullable(), // keep string from <input type="date">
  // ... other fields + validation
});

type Props = {
  open: boolean;
  onOpenChange: (v: boolean) => void;
  treatmentId?: number;           // if present → edit mode
  initialTreatment?: Treatment;   // optional: pass pre-fetched data to skip duplicate fetch
  onSaved?: () => void;           // callback after success
};

export default function TreatmentEditDialog({ open, onOpenChange, treatmentId, initialTreatment, onSaved }: Props) {
  const [confirmDateOpen, setConfirmDateOpen] = useState(false);
  const lastStatusRef = useRef<string | undefined>(undefined);

  // Replace these with your actual hooks
  const providers: Provider[] = useProviders() ?? [];
  const treatment: Treatment | undefined = useTreatment(treatmentId, initialTreatment);

  const normalizedProviders = useMemo(
    () => providers.filter(p => p.active !== false).sort((a, b) => a.name.localeCompare(b.name)),
    [providers]
  );

  const form = useForm<z.infer<typeof schema>>({
    resolver: zodResolver(schema),
    defaultValues: {
      providerId: undefined,
      invoiceStatus: "open",
      paymentDate: undefined,
    },
  });

  // Hydrate form when treatment/providers are ready
  useEffect(() => {
    if (!open) return;
    if (!treatment || !normalizedProviders.length) return;

    // Prefer providerId; fallback by actingProvider name
    let providerId: number | undefined = treatment.providerId ?? undefined;
    if (!providerId && treatment.actingProvider) {
      const match = normalizedProviders.find(
        p => p.name.trim().toLowerCase() === treatment.actingProvider!.trim().toLowerCase()
      );
      providerId = match?.id;
    }

    form.reset({
      providerId,
      invoiceStatus: (treatment.invoiceStatus as any) ?? "open",
      paymentDate: treatment.paymentDate ?? undefined,
      // ... map other fields
    });

    lastStatusRef.current = (treatment.invoiceStatus as any) ?? "open";
  }, [open, treatment, normalizedProviders, form]);

  // Watch invoiceStatus → open payment-date prompt when changing from open→paid/closed
  const status = form.watch("invoiceStatus");
  useEffect(() => {
    if (!open) return;
    const prev = lastStatusRef.current;
    if (!prev) {
      lastStatusRef.current = status;
      return;
    }
    // Only prompt on *change*, not on initial load/reset
    if (prev === "open" && (status === "paid" || status === "closed")) {
      setConfirmDateOpen(true);
    }
    lastStatusRef.current = status;
  }, [status, open]);

  const onSubmit = form.handleSubmit(async (values) => {
    const payload: Treatment = {
      providerId: values.providerId ?? null,
      invoiceStatus: values.invoiceStatus,
      paymentDate: values.paymentDate ?? null,
      // ... include other fields
    };

    const method = treatmentId ? "PATCH" : "POST";
    const url = treatmentId ? `/api/treatments/${treatmentId}` : `/api/treatments`;
    const res = await fetch(url, {
      method,
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(payload),
    });
    const json = await res.json();
    if (!json.ok) {
      // show toast json.message
      return;
    }
    onOpenChange(false);
    onSaved?.();
  });

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-xl">
        <DialogHeader>
          <DialogTitle>{treatmentId ? "Update Treatment" : "Create Treatment"}</DialogTitle>
        </DialogHeader>

        <Form {...form}>
          <form id="treatment-form" onSubmit={onSubmit} className="space-y-4">
            {/* Provider Select */}
            <FormField
              control={form.control}
              name="providerId"
              render={({ field }) => (
                <FormItem>
                  <FormLabel className="text-sm font-medium text-gray-700">Provider</FormLabel>
                  <Select
                    value={field.value ? String(field.value) : undefined}
                    onValueChange={(v) => field.onChange(v ? Number(v) : undefined)}
                    disabled={!normalizedProviders.length}
                  >
                    <FormControl>
                      <SelectTrigger className="mt-1">
                        <SelectValue placeholder="Select provider" />
                      </SelectTrigger>
                    </FormControl>
                    <SelectContent>
                      {normalizedProviders.map((p) => (
                        <SelectItem key={p.id} value={String(p.id)}>
                          {p.name}
                        </SelectItem>
                      ))}
                    </SelectContent>
                  </Select>
                  <FormMessage />
                </FormItem>
              )}
            />

            {/* Invoice Status + payment date */}
            <div className="grid grid-cols-2 gap-3">
              <FormField
                control={form.control}
                name="invoiceStatus"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel className="text-sm font-medium text-gray-700">Invoice Status</FormLabel>
                    <Select
                      value={field.value}
                      onValueChange={field.onChange}
                    >
                      <FormControl>
                        <SelectTrigger className="mt-1">
                          <SelectValue />
                        </SelectTrigger>
                      </FormControl>
                      <SelectContent>
                        <SelectItem value="open">Open</SelectItem>
                        <SelectItem value="paid">Paid</SelectItem>
                        <SelectItem value="closed">Closed</SelectItem>
                      </SelectContent>
                    </Select>
                    <FormMessage />
                  </FormItem>
                )}
              />

              <FormField
                control={form.control}
                name="paymentDate"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel className="text-sm font-medium text-gray-700">Payment Date</FormLabel>
                    <FormControl>
                      <Input type="date" value={field.value ?? ""} onChange={e => field.onChange(e.target.value || undefined)} />
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}
              />
            </div>

            {/* … other fields … */}

            <div className="flex justify-end gap-2">
              <Button variant="ghost" type="button" onClick={() => onOpenChange(false)}>Cancel</Button>
              <Button type="submit" form="treatment-form">
                {treatmentId ? "Update Treatment" : "Create Treatment"}
              </Button>
            </div>
          </form>
        </Form>
      </DialogContent>

      {/* Simple inline “enter payment date” prompt */}
      {confirmDateOpen && (
        <Dialog open={confirmDateOpen} onOpenChange={setConfirmDateOpen}>
          <DialogContent className="max-w-sm">
            <DialogHeader>
              <DialogTitle>Enter payment date</DialogTitle>
            </DialogHeader>
            <div className="space-y-3">
              <Input
                type="date"
                value={form.getValues("paymentDate") ?? ""}
                onChange={(e) => form.setValue("paymentDate", e.target.value || undefined)}
              />
              <div className="flex justify-end gap-2">
                <Button variant="ghost" onClick={() => setConfirmDateOpen(false)}>Skip</Button>
                <Button onClick={() => setConfirmDateOpen(false)}>Save</Button>
              </div>
            </div>
          </DialogContent>
        </Dialog>
      )}
    </Dialog>
  );
}

// stub hooks so this file compiles in isolation; replace with your real hooks.
function useProviders(): Provider[] {
  // return data from /api/providers
  return (window as any).__providers ?? [];
}
function useTreatment(id?: number, fallback?: Treatment): Treatment | undefined {
  return fallback;
}
